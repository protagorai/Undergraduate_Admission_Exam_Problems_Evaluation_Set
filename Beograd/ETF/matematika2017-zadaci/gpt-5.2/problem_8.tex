\section*{Zadatak 8}
Ako se razvije izraz
\[
2x\left(1+\sqrt[3]{x}\right)^{10}
 + x^4\left(x^2+\frac{1}{\sqrt[5]{x^2}}\right)^{12}
 + 5x^2\left(\sqrt[6]{x}-\frac{1}{\sqrt[6]{x}}\right)^{18}\qquad (x\in\mathbb{R}^+),
\]
i jedan od sabiraka je oblika $A\cdot x^4$, odrediti $A$.

\subsection*{Resenje}
Trazi se koeficijent uz $x^4$ u zbiru.

\paragraph{1) $2x(1+x^{1/3})^{10}$}
\[
2x\sum_{k=0}^{10}\binom{10}{k}x^{k/3}
=2\sum_{k=0}^{10}\binom{10}{k}x^{1+k/3}.
\]
Uslov $1+k/3=4$ daje $k=9$, pa doprinos iznosi
\[
2\binom{10}{9}=20.
\]

\paragraph{2) $x^4(x^2+x^{-2/5})^{12}$}
\[
x^4\sum_{k=0}^{12}\binom{12}{k}(x^2)^{12-k}(x^{-2/5})^{k}
=x^4\sum_{k=0}^{12}\binom{12}{k}x^{24-(12/5)k}.
\]
Za stepen $x^4$ treba $28-(12/5)k=4$, tj. $k=10$.
Doprinos je
\[
\binom{12}{10}=66.
\]

\paragraph{3) $5x^2(x^{1/6}-x^{-1/6})^{18}$}
\[
5x^2\sum_{k=0}^{18}\binom{18}{k}(x^{1/6})^{18-k}(-x^{-1/6})^k
 =5x^2\sum_{k=0}^{18}\binom{18}{k}(-1)^k x^{3-k/3}.
\]
Za stepen $x^4$ treba $5-k/3=4$, tj. $k=3$.
Doprinos je
\[
5\binom{18}{3}(-1)^3=-5\cdot 816=-4080.
\]

\paragraph{Ukupno}
\[
A=20+66-4080=-3994.
\]

\subsection*{Odgovor}
\[
\boxed{(A)\;\; -3994}
\]
