\section*{Zadatak 9}
Data je funkcija
\[
f(x)=
\begin{cases}
x^2, & x\ \text{je paran broj},\\[4pt]
\dfrac12(x-3), & x\ \text{je neparan broj}.
\end{cases}
\]
Odrediti vrednost izraza $f\bigl(f\bigl(f(17)+1\bigr)+1\bigr)$.

\subsection*{Resenje}
Kako je $17$ neparan,
\[
f(17)=\frac12(17-3)=7.
\]
Zatim
\[
f(17)+1=8 \quad\Rightarrow\quad f(8)=8^2=64.
\]
Dalje,
\[
f(8)+1=65 \quad\Rightarrow\quad f(65)=\frac12(65-3)=31.
\]

\subsection*{Odgovor}
\[
\boxed{(C)\;\; 31}
\]
