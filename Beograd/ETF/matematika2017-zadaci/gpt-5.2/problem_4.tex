\section*{Zadatak 4}
Neka su $z_1=2017+2018i$, $z_2=2018+2019i$ i
\[
w=(z_2-z_1)^{-2020}\cdot\left(\frac{\overline{z_1}+1}{2018}\right)^{2021}.
\]
Odrediti $|w|$.

\subsection*{Resenje}
Imamo
\[
z_2-z_1=(2018-2017)+(2019-2018)i=1+i,\qquad |1+i|=\sqrt{2}.
\]
Dalje,
\[
\overline{z_1}+1=(2017-2018i)+1=2018-2018i=2018(1-i),
\]
pa je
\[
\left|\frac{\overline{z_1}+1}{2018}\right|=|1-i|=\sqrt{2}.
\]
Zato
\[
|w|=|z_2-z_1|^{-2020}\cdot\left|\frac{\overline{z_1}+1}{2018}\right|^{2021}
=(\sqrt{2})^{-2020}\cdot(\sqrt{2})^{2021}=\sqrt{2}.
\]

\subsection*{Odgovor}
\[
\boxed{(D)\;\; \sqrt{2}}
\]
