\section*{Zadatak 2}
Ako je $V$ zapremina lopte, odrediti povrsinu lopte.

\subsection*{Resenje}
Za poluprecnik $r$ vazi
\[
V=\frac{4}{3}\pi r^3 \quad\Longrightarrow\quad r=\left(\frac{3V}{4\pi}\right)^{1/3}.
\]
Povrsina lopte je
\[
P=4\pi r^2
=4\pi\left(\frac{3V}{4\pi}\right)^{2/3}
=\left(36\pi V^2\right)^{1/3}.
\]

\subsection*{Odgovor}
\[
\boxed{(A)\;\; \sqrt[3]{36\pi V^2}}
\]
