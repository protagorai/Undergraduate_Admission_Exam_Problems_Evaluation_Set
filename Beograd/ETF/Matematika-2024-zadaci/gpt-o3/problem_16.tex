\section*{Problem 16}
Izračunati graničnu vrednost
\[
\lim_{x\to0}\frac{e^{\sin^{2}x}-\cos x}{\sqrt{x^{2}+1}-1}.
\]

\subsection*{Rešenje}
Koristimo razvoj u red oko $x=0$.
\begin{align*}
\sin^{2}x &= x^{2}-\tfrac{1}{3}x^{4}+\mathcal O(x^{6}),\\[4pt]
e^{\sin^{2}x} &= 1+x^{2}+\tfrac12 x^{4}+\mathcal O(x^{6}),\\[4pt]
\cos x &= 1-\tfrac12x^{2}+\tfrac{1}{24}x^{4}+\mathcal O(x^{6}).
\end{align*}
Dakle brojilac je
\[
e^{\sin^{2}x}-\cos x = \Bigl(1+x^{2}+\tfrac12x^{4}\Bigr)-\Bigl(1-\tfrac12x^{2}+\tfrac{1}{24}x^{4}\Bigr)=\tfrac32x^{2}+\mathcal O(x^{4}).\]

Za imenilac koristimo binomsku formulu:
\[
\sqrt{1+x^{2}}=1+\tfrac12x^{2}-\tfrac18x^{4}+\mathcal O(x^{6})\;\Longrightarrow\;\sqrt{1+x^{2}}-1=\tfrac12x^{2}+\mathcal O(x^{4}).
\]

Stoga
\[
\frac{e^{\sin^{2}x}-\cos x}{\sqrt{1+x^{2}}-1}=\frac{\tfrac32x^{2}+\mathcal O(x^{4})}{\tfrac12x^{2}+\mathcal O(x^{4})}\longrightarrow3.
\]
