\section*{Problem 3}
Compute the value of:
\[
49^{1-\log_7 2} + 5^{-\log_5 4}
\]

\subsection*{Solution}
\textbf{First term:}
\[
49^{1-\log_7 2} = 49^1 \cdot 49^{-\log_7 2} = 49 \cdot (7^2)^{-\log_7 2}
\]
\[
= 49 \cdot 7^{-2\log_7 2} = 49 \cdot 7^{\log_7 2^{-2}} = 49 \cdot 2^{-2} = 49 \cdot \frac{1}{4} = \frac{49}{4}
\]

\textbf{Second term:}
\[
5^{-\log_5 4} = 5^{\log_5 4^{-1}} = 5^{\log_5 \frac{1}{4}} = \frac{1}{4}
\]

\textbf{Sum:}
\[
\frac{49}{4} + \frac{1}{4} = \frac{50}{4} = \frac{25}{2}
\]

\subsection*{Answer}
$\displaystyle\frac{25}{2}$ (option \textbf{B}).
