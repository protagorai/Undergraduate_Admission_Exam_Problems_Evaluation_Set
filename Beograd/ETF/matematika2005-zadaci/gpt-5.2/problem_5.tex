\section*{Problem 5}
Ako su $x_1$ i $x_2$ rešenja jednačine $x^2+x-2005=0$, tada je
\[
2x_1^2+x_1x_2+x_2^2+x_1-2005
\]
jednako:
\[
\text{(A) }2005\quad \text{(B) }2006\quad \text{(C) }2007\quad \text{(D) }2008\quad \text{(E) }2009\quad \text{(N) Ne znam.}
\]

\subsection*{Solution}
Za korene važi $x_1+x_2=-1$ i $x_1x_2=-2005$. Takođe,
\[
x_1^2+x_2^2=(x_1+x_2)^2-2x_1x_2=1-2(-2005)=4011.
\]
Traženi izraz je
\begin{align*}
2x_1^2+x_1x_2+x_2^2+x_1-2005
&=(x_1^2+x_2^2)+x_1^2+x_1x_2+x_1-2005\\
&=4011+\bigl(x_1^2+x_1\bigr)+(-2005)-2005\\
&=4011+\bigl(x_1^2+x_1\bigr)-4010.
\end{align*}
Pošto $x_1$ zadovoljava $x_1^2+x_1-2005=0$, sledi $x_1^2+x_1=2005$, pa je izraz
\[
4011+2005-4010=2006.
\]

\subsection*{Answer}
$2006$ (option \textbf{B}).
