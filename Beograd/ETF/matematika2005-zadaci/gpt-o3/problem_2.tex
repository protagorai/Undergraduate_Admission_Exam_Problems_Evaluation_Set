\section*{Problem 2}
Za funkciju $f(x)=x^{2}+x+1$ izračunati
\[
f(x+2)-2f(x+1)+f(x).
\]

\subsection*{Rešenje}
\begin{align*}
f(x+2)&=(x+2)^2+(x+2)+1=x^2+5x+7,\\
f(x+1)&=(x+1)^2+(x+1)+1=x^2+3x+3,\\
f(x)&=x^2+x+1.
\end{align*}
Zbir postaje
\[
(x^2+5x+7)-2(x^2+3x+3)+(x^2+x+1)=2.
\]

\subsection*{Odgovor}
$2$ (opcija \textbf{A}).
