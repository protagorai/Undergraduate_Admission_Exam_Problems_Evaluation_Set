\section*{Problem 1}
Izračunati vrednost
\[
\frac{\sqrt{32}}{\sqrt{8}}+\sqrt{3}\,\sqrt{12}-\sqrt{50}\,\sqrt{2}.
\]

\subsection*{Rešenje}
\[
\frac{\sqrt{32}}{\sqrt{8}}=\frac{4\sqrt2}{2\sqrt2}=2,\qquad
\sqrt{3}\,\sqrt{12}=\sqrt{36}=6,\qquad
\sqrt{50}\,\sqrt{2}=5\sqrt2\cdot\sqrt2=10.
\]
Dakle
\[
2+6-10=-2.
\]

\subsection*{Odgovor}
$-2$ (opcija \textbf{B}).
