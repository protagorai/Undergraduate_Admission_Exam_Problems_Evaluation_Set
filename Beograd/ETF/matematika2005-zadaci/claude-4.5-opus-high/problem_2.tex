\section*{Problem 2}
If $f(x) = x^2 + x + 1$, find $f(x+2) - 2f(x+1) + f(x)$ for every $x$.

\subsection*{Solution}
Let's compute each term:

\textbf{Compute $f(x+2)$:}
\[
f(x+2) = (x+2)^2 + (x+2) + 1 = x^2 + 4x + 4 + x + 2 + 1 = x^2 + 5x + 7
\]

\textbf{Compute $f(x+1)$:}
\[
f(x+1) = (x+1)^2 + (x+1) + 1 = x^2 + 2x + 1 + x + 1 + 1 = x^2 + 3x + 3
\]

\textbf{Compute $f(x)$:}
\[
f(x) = x^2 + x + 1
\]

\textbf{Calculate $f(x+2) - 2f(x+1) + f(x)$:}
\begin{align*}
&= (x^2 + 5x + 7) - 2(x^2 + 3x + 3) + (x^2 + x + 1) \\
&= x^2 + 5x + 7 - 2x^2 - 6x - 6 + x^2 + x + 1 \\
&= (x^2 - 2x^2 + x^2) + (5x - 6x + x) + (7 - 6 + 1) \\
&= 0 + 0 + 2 \\
&= 2
\end{align*}

\subsection*{Answer}
$2$ (option \textbf{A}).
