% Problem 6 – Matematički prijemni 2023
\begin{solution}
Jednačina ima oblik
\[
 p x^{3}+q x^{2}+p x+1=0\qquad(p\ne0).
\]
Podelimo са \(p\) (да добијемо монични полином):
\[
 x^{3}+\frac{q}{p}x^{2}+x+\frac1p=0.
\]
Нека су \(x_{1},x_{2},x_{3}\) корени.  За монични кубни полином важи
\[
\begin{aligned}
x_{1}+x_{2}+x_{3}&=-\frac{q}{p},\\[2pt]
x_{1}x_{2}+x_{2}x_{3}+x_{1}x_{3}&=1,\\[2pt]
x_{1}x_{2}x_{3}&=-\frac1p.
\end{aligned}
\]
Тражени израз је
\[
\frac1{x_{1}}+\frac1{x_{2}}+\frac1{x_{3}}=
\frac{x_{2}x_{3}+x_{1}x_{3}+x_{1}x_{2}}{x_{1}x_{2}x_{3}}
  =\frac{1}{-\dfrac1p}=\boxed{-p},
\]
што одговара опцији \textbf{E}. \qedhere
\end{solution}
