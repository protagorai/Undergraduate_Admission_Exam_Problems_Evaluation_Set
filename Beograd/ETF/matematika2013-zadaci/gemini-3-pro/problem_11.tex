\[ x_1^2 + x_2^2 = (x_1 + x_2)^2 - 2x_1x_2 = (-b)^2 - 2c = b^2 - 2c \]

\[ (x_1^2 + x_2^2)^2 = (b^2 - 2c)^2 \]

\[ x_1^4 + x_2^4 = (b^4 - 4b^2c + 4c^2) - 2x_1^2x_2^2 \]
