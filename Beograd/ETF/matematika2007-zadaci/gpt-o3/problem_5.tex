\section*{Problem 5}
Neka je $|x+a|=a$ ($a\ge0$). Izračunati
\[
\bigl|\,|x|-a\bigr|.
\]

\subsection*{Rešenje}
Jednačina $|x+a|=a$ ima dva rešenja.
\begin{align*}
\text{1)}\;x+a&=a &\Rightarrow&& x&=0,\\[2pt]
\text{2)}\;x+a&=-a &\Rightarrow&& x&=-2a.
\end{align*}
Za oba dobijena $x$ izračunajmo traženi izraz.
\begin{itemize}
 \item Ako je $x=0$: $|x|=0$ i $|0-a|=a$.
 \item Ako je $x=-2a$: $|x|=2a$ i $|2a-a|=a$.
\end{itemize}
Dakle, uvek važi $\bigl|\,|x|-a\bigr|=a$.
\subsection*{Odgovor}
$a$ (opcija \textbf{B}).
