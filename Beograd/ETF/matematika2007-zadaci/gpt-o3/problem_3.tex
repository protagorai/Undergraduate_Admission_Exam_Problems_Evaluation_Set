\section*{Problem 3}
Odrediti rastojanje tačke $(1,-1)$ od prave
\[
 x+2y-4=0.
\]

\subsection*{Rešenje}
Za rastojanje tačke $(x_0,y_0)$ od prave $Ax+By+C=0$ važi
\[
 d=\frac{|Ax_0+By_0+C|}{\sqrt{A^{2}+B^{2}}}.
\]
Ovde su $A=1$, $B=2$, $C=-4$ i $(x_0,y_0)=(1,-1)$, pa
\[
|Ax_0+By_0+C|=|1\cdot1+2\cdot(-1)-4|=|-5|=5,\qquad
\sqrt{A^{2}+B^{2}}=\sqrt{1+4}=\sqrt5.
\]
Otuda je
\[
 d=\frac5{\sqrt5}=\sqrt5.
\]
\subsection*{Odgovor}
$\sqrt5$ (opcija \textbf{E}).
