\section*{Problem 2}
Izračunati vrednost izraza
\[
\left[\left(1+\frac12\right)^{-1}:\left(1+\frac13\right)\right]^{-2}\cdot\left(1+\frac14\right).
\]

\subsection*{Solution}
Najpre:
\[
\left(1+\frac12\right)^{-1}=\left(\frac32\right)^{-1}=\frac23,\qquad
1+\frac13=\frac43.
\]
Zato je količnik u uglastoj zagradi
\[
\frac23:\frac43=\frac23\cdot\frac34=\frac12.
\]
Zatim
\[
\left(\frac12\right)^{-2}=4,
\qquad
1+\frac14=\frac54,
\]
pa je ceo izraz
\[
4\cdot\frac54=5.
\]

\subsection*{Answer}
$5$ (option \textbf{A}).
