\section*{Problem 12}
If $x\ne 0$ and
\[
f(x)-2f\!\left(\frac{1}{x}\right)=x^2,
\]
find $f(2)$.

\subsection*{Solution}
Write the equation for $x$ and for $\frac{1}{x}$:
\[
f(x)-2f\!\left(\frac{1}{x}\right)=x^2 \qquad (1)
\]
\[
f\!\left(\frac{1}{x}\right)-2f(x)=\frac{1}{x^2} \qquad (2)
\]
Multiply (2) by $2$ and add to (1):
\[
\bigl(f(x)-4f(x)\bigr)+\bigl(-2f(1/x)+2f(1/x)\bigr)=x^2+\frac{2}{x^2},
\]
so
\[
-3f(x)=x^2+\frac{2}{x^2}\Rightarrow
f(x)=-\frac{x^2+\frac{2}{x^2}}{3}.
\]
Hence
\[
f(2)=-\frac{4+\frac{2}{4}}{3}=-\frac{4+\frac12}{3}=-\frac{9/2}{3}=-\frac{3}{2}.
\]

\subsection*{Answer}
$-\dfrac{3}{2}$ (option \textbf{Ð”}).
